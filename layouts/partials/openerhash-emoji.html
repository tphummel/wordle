{{/* Convert an opener hash string (e.g. "AAPAA") into its emoji representation. The taxonomy stores hashes using the letters: - C: correct (green) - P: present (yellow) - A: absent (black) Some older data may include placeholder "X" characters; treat those as blanks. */}} {{- $hash := printf "%s" . -}} {{- $tiles := slice -}} {{- range (split $hash "") -}} {{- $emoji := "" -}} {{- $char := upper . -}} {{- if eq $char "C" -}} {{- $emoji = "🟩" -}} {{- else if eq $char "P" -}} {{- $emoji = "🟨" -}} {{- else if or (eq $char "A") (eq $char "X") -}} {{- $emoji = "⬛️" -}} {{- else -}} {{- $emoji = $char -}} {{- end -}} {{- $tiles = $tiles | append $emoji -}} {{- end -}} {{- return (delimit $tiles "") -}}
